#use 5.010;

#-------------------------------
#Checks to see if string has pattern: 
#  [WS]<realnumber>[WS]
#-------------------------------
sub real_num_match {
   my $string;
   $string = $_[0];
   if ( $string =~ /^([\t ]*)(-\d+)?(\d*\.?\d+)?(e-?\d+)?[\t ]*$/i )
        { return 1; }
   else { return -1; }

}

#-------------------------------------------
#Checks to see if string has pattern:
# [AWS] <string [AWS] = [AWS] string> [AWS]
# This is useful for reading parameter file
#-------------------------------------------

sub str_eq_match {
   my $string;
   $string = $_[0];
   if ( $string =~ /(\s*\b\w\b\s*=\s*\b\w\b\s*)/ )
      {return 1; }
   else    {return -1; }
}
